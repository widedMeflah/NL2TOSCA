tosca_definitions_version: tosca_simple_yaml_1_0

imports:
  - indigo_custom_types: https://raw.githubusercontent.com/indigo-dc/tosca-types/master/custom_types.yaml

description: TOSCA template for deploying Wine Cellar Application with Apache and MySQL

topology_template:
  inputs:
    db_name:
      type: string
      default: winecellar
    db_user:
      type: string
      default: wineuser
    db_password:
      type: string
      default: winepwd
    db_root_password:
      type: string
      default: rootpwd

  node_templates:
    web_server:
      type: tosca.nodes.INDIGO.Compute
      capabilities:
        endpoint:
          properties:
            network_name: PUBLIC
            ports:
              http_port:
                protocol: tcp
                source: 80
              https_port:
                protocol: tcp
                source: 443
        host:
          properties:
            num_cpus: 1
            mem_size: 1 GB
        os:
          properties:
            type: linux
            distribution: ubuntu

    apache:
      type: tosca.nodes.WebServer.Apache
      requirements:
        - host: web_server

    wine_app:
      type: tosca.nodes.WebApplication
      requirements:
        - host: apache
        - database_endpoint: mysql_database
      interfaces:
        Standard:
          create:
            implementation: scripts/install_app.sh
            inputs:
              github_url: https://github.com/wine-cellar-app/php-version
              db_host: { get_attribute: [ mysql_server, private_address ] }
              db_name: { get_property: [ mysql_database, name ] }
              db_user: { get_property: [ mysql_database, user ] }
              db_password: { get_property: [ mysql_database, password ] }

    mysql_server:
      type: tosca.nodes.INDIGO.Compute
      capabilities:
        endpoint:
          properties:
            network_name: PRIVATE
        host:
          properties:
            num_cpus: 1
            mem_size: 4 GB
            disk_size: 10 GB
        os:
          properties:
            type: linux
            distribution: ubuntu

    mysql_database:
      type: tosca.nodes.Database.MySQL
      properties:
        name: { get_input: db_name }
        user: { get_input: db_user }
        password: { get_input: db_password }
        root_password: { get_input: db_root_password }
      requirements:
        - host: mysql_server
      interfaces:
        Standard:
          configure:
            implementation: scripts/configure_db.sh
            inputs:
              db_url: http://example.com/wine_cellar.sql

  outputs:
    website_url:
      value: { concat: [ 'http://', get_attribute: [ web_server, public_address ] ] }
    database_endpoint:
      value: { get_attribute: [ mysql_server, private_address ] }
